            Login:
                            POST /users para criar
                            ex:{
                               	"user":{
                                   "email": "coiso@gmail.com",
                                   "password": "notblank",
                                   "password_confirmation": "notblank"
                               	}
                               }
                               
                            POST /users/login retorn auth_token
                             ex:{
                                  "email": "coiso@gmail.com",
                                  "password": "notblank"
                                }

            Pagination:
                            GET /companies?per_page=x
                                /companies?per_page=x&page=y
                                x = quantidade por pagina
                                y = nr de pagina
